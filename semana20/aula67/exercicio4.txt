exports.handler = async (event) => {
    if (!event.text || event.text.indexOf("@") === -1) {
           return {
               "isEmail": false,
               "reason": "Não contém @"
           }
    }

    if (!event.text || event.text.indexOf(".") === -1) {
           return {
               "isEmail": false,
               "reason": "Não contém ."
           }
    }

    return {
        "isEmail": true
    }
}
